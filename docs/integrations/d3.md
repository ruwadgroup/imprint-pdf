# Integration — D3

D3 isn't a React library, so this is a manual SVG-to-imprint pipeline. Render D3
to an SVG string server-side, then pass it to `<Svg>`.

## Pattern

```ts
import * as d3 from 'd3';
import { JSDOM } from 'jsdom'; // or use a virtual DOM in Node
```

```ts
function renderD3Chart(data: DataPoint[]): string {
  const width = 520;
  const height = 260;
  const margin = { top: 20, right: 20, bottom: 40, left: 50 };

  // Create a detached SVG DOM
  const dom = new JSDOM('<!DOCTYPE html><body></body>');
  const body = d3.select(dom.window.document.querySelector('body')!);

  const svg = body
    .append('svg')
    .attr('width', width)
    .attr('height', height)
    .attr('xmlns', 'http://www.w3.org/2000/svg');

  const x = d3
    .scaleTime()
    .domain(d3.extent(data, (d) => d.date) as [Date, Date])
    .range([margin.left, width - margin.right]);

  const y = d3
    .scaleLinear()
    .domain([0, d3.max(data, (d) => d.value)!])
    .nice()
    .range([height - margin.bottom, margin.top]);

  const line = d3
    .line<DataPoint>()
    .x((d) => x(d.date))
    .y((d) => y(d.value))
    .curve(d3.curveMonotoneX);

  // Grid lines
  svg
    .append('g')
    .attr('stroke', '#E5E7EB')
    .attr('stroke-dasharray', '3,3')
    .call(
      d3
        .axisLeft(y)
        .tickSize(-(width - margin.left - margin.right))
        .tickFormat(() => ''),
    )
    .attr('transform', `translate(${margin.left},0)`)
    .select('.domain')
    .remove();

  // Line path
  svg
    .append('path')
    .datum(data)
    .attr('fill', 'none')
    .attr('stroke', '#3B82F6')
    .attr('stroke-width', 2)
    .attr('d', line);

  // Axes
  svg
    .append('g')
    .attr('transform', `translate(0,${height - margin.bottom})`)
    .call(
      d3
        .axisBottom(x)
        .ticks(5)
        .tickFormat(d3.timeFormat('%b %y') as any),
    )
    .attr('font-size', '10')
    .attr('color', '#9CA3AF')
    .select('.domain')
    .attr('stroke', '#E5E7EB');

  return dom.window.document.querySelector('svg')!.outerHTML;
}
```

```tsx
// In your React component
const chartSvg = renderD3Chart(data);

<Svg src={chartSvg} className="w-full h-64" />;
```

## Alternative: `@nivo/core` / Observable Plot

Libraries with first-class SSR SVG output integrate more cleanly:

```ts
import { Plot } from '@observablehq/plot';

const chart = Plot.line(data, { x: 'date', y: 'value' }).plot({
  width: 520,
  height: 260,
  style: { fontFamily: 'Inter, sans-serif', fontSize: '10px' },
  document: new JSDOM('<!DOCTYPE html>').window.document,
});

const svgString = chart.outerHTML;
```

```tsx
<Svg src={svgString} className="w-full h-64" />
```
